public class Main {
    static int n=11;
    static boolean check(int mid){
        return mid<=7;
    }
    public static void main(String[] args){
        int l=0, r=n-1, ans=-1;
        while (l<=r){
            int mid=l+(r-l)/2;
            if (check(mid)){
                ans=mid;l=mid+1;
            }else r=mid-1;
        }

        System.out.println(ans);
    }
}
